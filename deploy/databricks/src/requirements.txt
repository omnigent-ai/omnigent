# Python dependencies for the Omnigent Databricks App.
#
# `deploy.py` builds the omnigent wheel (which bundles the prebuilt web-UI
# assets a source-only checkout can't produce), copies it into this directory,
# and rewrites the install line below to point at that wheel, e.g.:
#
#     ./omnigent-0.1.0-py3-none-any.whl[databricks]
#
# The `databricks` extra pulls the Databricks SDK (for minting Lakebase OAuth
# tokens) and the psycopg Postgres driver. If you publish omnigent to an index
# the app can reach, you can install from there instead of the local wheel.
omnigent[databricks]
